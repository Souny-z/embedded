## Pointer

1. 指针: 保存的是变量的地址

2. 基本指针

   ```C
   #include <stdio.h>
   
   int main (){
       int var = 14;
       int *p = &var;
       printf("%d\n",var); // 14
       printf("%s\n",p); // 打印P的内存地址 
   }
   ```

3. 指针与数组

   - 基本指针

     ```C
     int p; // 变量
     int *p; // 整形数据指针
     int p[3]; // 数组
     int *p[3]; // 指针数组，数组内存放的是指针
     int (*p)[3]; // 数组指针，为单个指针，该指针地址保存的是数组
     int **p;// 一个指针地址保存了另外的指针地址，另外的指针指向实际变量
     
     void *p;
     *(int *)p; // 将指针类型转换为int
     ```

   - 指针与一维数组

     ```C
     // 指针遍历数组
     void pointerArray(){
         int arr[5] = {1, 4, 5, 7, 8};
         int *p = &arr[0];
         for (size_t i = 0; i < sizeof(arr)/sizeof(int); i++)
         {
           printf("%d\n",*(p+i));
         }
     }
     // 倒序输出数组
     void pinterArrayReverse()
     {
       int arr[5] = {1, 4, 5, 7, 8};
       int n = sizeof(arr) / sizeof(int);
       int *p = &arr[n-1];
       for (size_t i = n; i > 0; i--)
       {
         printf("%d", *p--);
         puts("");
       }
     }
     ```

   - 指针与二维数组

     ```C
     // 指针与二维数组
     void pointerDoubleArray(){
         int arr[3][4] = {{11, 1, 14, 12}, {15, 22, 33, 44}, {25, 36, 47, 58}};
         int *p = &arr[0][0];
         printf("%d__%d\n", arr, **arr); // 6421936__11
         printf("%d__%d\n", arr, *(*arr + 1)); //外：6421936__1
         printf("%d__%d\n", arr + 1, *(*(arr + 1))); // 内：6421952__15
         printf("%d__%d\n", arr[0] + 1, *(*(arr + 1) + 1)); // 6421940__22
     }
     ```

4. 指针与函数

   - 传值调用与传址调用

      ```C
      // 传值调用 不改变实参
      int swap(int a, int b)
      {
        // static int r = 0;
        int temp = a;
        a = b;
        b = temp;
        // r = a;
        return a;
      }
      
      // 传址调用 改变实参
      void swap2(int *p, int *q)
      {
        int temp = 0;
        temp = *p;
        *p = *q;
        *q = temp;
      }
      ```

   - 指针函数

      ```C
      // 指针函数 函数标有指针符号，函数内部有指针，切返回值是指针
      # include <stdio.h>
      # include <stdlib.h>
      
      int * func_sum(int n)
      {
          static int sum = 0;
          // sum = sum + n;
          int *p = &sum;
          return p;
      }
      
      int main(void)
      {
          int *p = func_sum(10); 
          printf("sum:%d\n", *p);
          return 0;
      }
      
      ```

   - 函数指针

      ```C
      // 函数指针 通过指针调用函数
      int add(int a, int b){
          return a+b;
      }
      void main(){
          int (*p) (int, int) = &add;
          int res = p(1,2); // 返回1+2的值
      }
      ```
